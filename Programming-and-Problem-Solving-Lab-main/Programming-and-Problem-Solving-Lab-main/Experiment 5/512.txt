5.1.2


marks = list(map(int, input().split()))


total = sum(marks)


aggregate = total / 4


if aggregate > 75:
    grade = "Distinction"
elif 60 <= aggregate < 75:
    grade = "First Division"
elif 50 <= aggregate < 60:
    grade = "Second Division"
elif 40 <= aggregate < 50:
    grade = "Third Division"
else:
    grade = "Fail"

print(total)
print(f"{aggregate:.2f}")
print(grade)
